68  public:
69 
70  String name;
71  String category; // module
72  unsigned int parent = 0; // 0 is no parent
73 
74  inline Frame()
75  {
76  }

78  inline Frame(const String& _name, const String& _category, unsigned int _parent)
79  : name(_name), category(_category), parent(_parent)
80  {
81  }
82 
83  inline bool operator<(const Frame& compare) const noexcept
84  {
85  if (parent < compare.parent) {
86  return true;
87  } else if (parent > compare.parent) {
88  return false;
89  }
90  auto n = name.compareTo(compare.name);
91  if (n < 0) {
92  return true;
93  } else if (n > 0) {
94  return false;
95  }
96  // TAG: compare reference pointer value
97  auto c = category.compareTo(compare.category); // likely to be equal
98  if (c < 0) {
99  return true;
100  }
101  return false;
102  }
103 
104  inline bool operator==(const Frame& compare) const noexcept
105  {
106  return (parent == compare.parent) &&
107  (name == compare.name) &&
108  (category == compare.category);
109  }
