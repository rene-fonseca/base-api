83  class Frame {
84  public:
85 
86  String name;
87  String category; // module
88  unsigned int parent = 0; // 0 is no parent
89 
90  inline Frame()
91  {
92  }
93 
94  inline Frame(const String& _name, const String& _category, unsigned int _parent)
95  : name(_name), category(_category), parent(_parent)
96  {
97  }
98 
99  inline bool operator<(const Frame& compare) const noexcept
100  {
101  if (parent < compare.parent) {
102  return true;
103  } else if (parent > compare.parent) {
104  return false;
105  }
106  auto n = name.compareTo(compare.name);
107  if (n < 0) {
108  return true;
109  } else if (n > 0) {
110  return false;
111  }
112  // TAG: compare reference pointer value
113  auto c = category.compareTo(compare.category); // likely to be equal
114  if (c < 0) {
115  return true;
116  }
117  return false;
118  }
119 
120  inline bool operator==(const Frame& compare) const noexcept
121  {
122  return (parent == compare.parent) &&
123  (name == compare.name) &&
124  (category == compare.category);
125  }
126  };
