57  inline FileRegion fixRegion(const FileRegion& region) noexcept
58  {
59  unsigned int index = region.getOffset() % granularity;
60  long long offset = region.getOffset()/granularity*granularity;
61  unsigned int minimumSize = region.getSize() + index;
62  unsigned int size = (maximum(minimumSize, preferredWindowSize) + granularity - 1)/granularity*granularity;
63  long long availableBytes = fileSize - offset;
64  if ((availableBytes > 0) && (size > availableBytes)) {
65  size = static_cast<unsigned int>(availableBytes);
66  }
67  if (size < minimumSize) { // force IOException if minimumSize bytes is not available
68  size = minimumSize;
69  }
70  return FileRegion(offset, size);
71  }
