178  HashSetImpl(MemorySize capacity)
179  {
180  capacity = maximum(capacity, MINIMUM_CAPACITY);
181  capacity = minimum(capacity, MAXIMUM_CAPACITY);
182  log2OfCapacity = Math::iLog2(capacity);
183  if ((static_cast<MemorySize>(1U) << log2OfCapacity) < capacity) {
184  ++log2OfCapacity;
185  capacity *= 2;
186  }
187  this->capacity = capacity;
188  mask = capacity - 1;
189  size = 0;
190  table.setSize(capacity);
191  fill<Node*>(table.getElements(), table.getSize(), 0);
192  }

197  inline HashSetImpl(const HashSetImpl& copy)
198  : table(copy.capacity),

200  mask(copy.mask),
201  log2OfCapacity(copy.log2OfCapacity),
202  size(copy.size)
203  {
204  table.setSize(capacity); // entries set below
205 
206  // copy all buckets
207  const Node* const* src = copy.getBuckets();
208  const Node* const* end = src + copy.capacity;
209  Node** dest = getBuckets();
210  while (src != end) {
211  const Node* srcNode = *src++;
212  if (srcNode) {

214  *dest++ = firstNode;

216  srcNode = srcNode->getNext();
217  while (srcNode) {

219  destNode = destNode->getNext();
220  srcNode = srcNode->getNext();

231  void grow()
232  {
233  if (capacity < MAXIMUM_CAPACITY) {
234  capacity *= 2;
235  mask = capacity - 1;
236  table.setSize(capacity); // upper entries set below
237  ++log2OfCapacity;
238 
239  // unfold buckets
240  Node** lowerBucket = getBuckets();
241  Node** upperBucket = lowerBucket + capacity/2;
242  const Node* const* endBucket = upperBucket;
243  const MemorySize bitMask = capacity/2; // the bit that desides the half
244  while (lowerBucket != endBucket) {
245  Node* srcNode = *lowerBucket;
246  if (srcNode) {
247  // keep order and do NOT construct or destroy nodes
248  if ((srcNode->getHash() & bitMask) == 0) {
249  // move from first to second bucket
250  Node* parentSrcNode = srcNode;
251  srcNode = srcNode->getNext();
252  while (srcNode && ((srcNode->getHash() & bitMask) == 0)) {
253  parentSrcNode = srcNode;
254  srcNode = srcNode->getNext();
255  }
256  if (srcNode) {
257  parentSrcNode->setNext(srcNode->getNext()); // unlink
258  *upperBucket = srcNode;
259  srcNode = srcNode->getNext();
260 
261  Node* destNode = *upperBucket;
262  // destNode->setNext(nullptr); // only set for last node
263 
264  while (srcNode) {
265  Node* nextNode = srcNode->getNext();
266  if (srcNode->getHash() & bitMask) {
267  parentSrcNode->setNext(nextNode); // unlink
268  destNode->setNext(srcNode);
269  destNode = destNode->getNext();
270  // destNode->setNext(nullptr); // only set for last node
271  }
272  parentSrcNode = srcNode;
273  srcNode = nextNode;
274  }
275  destNode->setNext(nullptr);
276  }
277  } else {
278  // move from second to first bucket
279  *upperBucket = *lowerBucket;
280  *lowerBucket = 0;
281 
282  Node* parentSrcNode = srcNode;
283  srcNode = srcNode->getNext();
284  while (srcNode && (srcNode->getHash() & bitMask)) {
285  parentSrcNode = srcNode;
286  srcNode = srcNode->getNext();
287  }
288  if (srcNode) {
289  parentSrcNode->setNext(srcNode->getNext()); // unlink
290  *lowerBucket = srcNode;
291  srcNode = srcNode->getNext();
292 
293  Node* destNode = *lowerBucket;
294  // destNode->setNext(nullptr); // only set for last node
295 
296  while (srcNode) {
297  Node* nextNode = srcNode->getNext();
298  if ((srcNode->getHash() & bitMask) == 0) {
299  parentSrcNode->setNext(nextNode); // unlink
300  destNode->setNext(srcNode);
301  destNode = destNode->getNext();
302  // destNode->setNext(nullptr); // only set for last node
303  }
304  parentSrcNode = srcNode;
305  srcNode = nextNode;
306  }
307  destNode->setNext(nullptr); // terminate linked list
308  }
309  }
310  } else {
311  *upperBucket = 0; // lower bucket already empty
312  }
313  ++lowerBucket;
314  ++upperBucket;
315  }
316  }
317  }

322  inline void shrink()
323  {
324  if (capacity > MINIMUM_CAPACITY) {
325  // fold upper half buckets into lower buckets
326  Node** lowerBucket = getBuckets();
327  Node** upperBucket = lowerBucket + capacity/2;
328  const Node* const* endBucket = upperBucket;
329  while (lowerBucket != endBucket) {
330  // merge buckets
331  if (!*upperBucket) {
332  // nothing to do
333  } else if (!*lowerBucket) {
334  *lowerBucket = *upperBucket;
335  } else {
336  // TAG: bucket could hold link to last node
337  Node* node = *lowerBucket;
338  while (node->getNext()) { // go to last node
339  node = node->getNext();
340  }
341  node->setNext(*upperBucket);
342  }
343  ++lowerBucket;
344  ++upperBucket;
345  }
346  capacity /= 2;
347  mask = capacity - 1;
348  --log2OfCapacity;
349  table.setSize(capacity);
350  }

372  bool hasValue(const Value& key) noexcept
373  {
374  const unsigned long hash = getHash(key);
375  Node** bucket = getBuckets() + (hash & mask);
376 
377  Node* parent = nullptr;
378  Node* child = *bucket;
379  if (child) {
380  if ((child->getHash() == hash) && (*child->getKey() == key)) {
381  return true; // this case should be fairly likely
382  }
383  parent = child;
384  child = child->getNext();
385  while (child &&
386  ((child->getHash() != hash) || (*child->getKey() != key))) {
387  parent = child;
388  child = child->getNext();
389  }
390  if (child) {
391  Node* temp = child->getNext();
392  Node* first = *bucket;
393  child->setNext(first); // cyclic loop to first
394  *bucket = child; // move to first - still cyclic loop
395  parent->setNext(temp); // detach child
396  }
397  }
398  return child;
399  }

424  void add(const Value& value)
425  {
426  const unsigned long hash = getHash(value);
427  Node** buckets = getBuckets() + (hash & mask);
428  if (*buckets) {
429  Node* parent = nullptr;
430  Node* child = *buckets;
431  while (child && ((child->getHash() != hash) || (child->getValue() != value))) {
432  parent = child;
433  child = child->getNext();
434  }
435  if (!child) { // if value not already in list
436  child = new Node(hash, value);
437  if (parent) {
438  parent->setNext(child);
439  }
440  ++size;
441  }
442  } else {
443  *buckets = new Node(hash, value);
444  ++size;
445  }
446  if (size > capacity) { // TAG: find simple rules that works
447  grow();
448  }
449  }

454  void remove(const Value& value)
455  {
456  const unsigned long hash = getHash(value);
457  Node** bucket = getBuckets() + (hash & mask);
458  Node* child = *bucket;
459  if (!child) {
460  throw InvalidNode(this);
461  }
462  if ((child->getHash() == hash) && (child->getValue() == value)) {
463  *bucket = child->getNext(); // unlink first node (next could be 0)
464  } else {
465  Node* parent = child;
466  child = child->getNext();
467  while (child && ((child->getHash() != hash) || (child->getValue() != value))) {
468  parent = child;
469  child = child->getNext();
470  }
471  if (!child) {
472  throw InvalidNode(this);
473  }
474  parent->setNext(child->getNext()); // unlink node from linked list
475  }
476  --size;
477  delete child;
478  if (size < capacity/2) { // TAG: find simple rules that works
479  shrink();
480  }
481  }
