327  auto node = new Node(nullptr, nullptr, nullptr, value);
328  this->elements = new typename BinaryTree<Value>::BinaryTreeImpl(node); // attach root node
329  return Pair<Node*, bool>(node, true);

331 
332  while (true) {
333  const int result = compare<const Key&>(value, node->getValue());
334  if (result < 0) {
335  if (auto left = node->getLeft()) {
336  node = left;
337  } else { // attach left child node
338  Node* newNode = new Node(node, nullptr, nullptr, value);
339  node->setLeft(newNode);

345  node = right;

437  {

444  if (node == getRoot()) { // set new root
445  if (auto left = node->getLeft()) {
446  BinaryTree<Value>::setRoot(left);
447  } else if (auto right = node->getRight()) {
448  BinaryTree<Value>::setRoot(right);
449  }
450  }
451 
452  while (true) {
453  if (node->getLeft()) {
454  this->elements->rotateRight(node);
455  } else if (node->getRight()) {
456  this->elements->rotateLeft(node);
457  } else {
458  break; // stop when the node is a leaf in the tree
459  }
460  }
461 
462  Node* parent = node->getParent();
463  if (parent) {
464  if (parent->getLeft() == node) {
465  // node if left child node of parent
466  parent->setLeft(nullptr); // detach node
467  } else {
468  // node if right child node of parent
469  parent->setRight(nullptr); // detach node
470  }
471  delete node;

505  {
