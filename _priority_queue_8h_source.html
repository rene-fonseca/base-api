99  {
100  auto node = elements.find(Node(priority));
101  if (node) { // does the priority already exist in the tree
102  auto& nodeValue = node->getValue();
103  Queue<Value>& queue = nodeValue.getValue();
104  queue.push(value);
105  } else {
106  Queue<Value> queue;
107  queue.push(value);
108  elements.add(Node(priority, queue));
109  }
110  ++size; // always increment 'cause we always push a value onto some queue
