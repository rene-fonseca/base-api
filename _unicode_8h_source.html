198  static inline MemorySize writeUTF8(uint8* dest, ucs4 ch) noexcept
199  {
200  if (ch <= 0x7f) { // 7 bit
201  *dest++ = static_cast<uint8>(ch);
202  return 1;
203  }
204  if (ch <= 0x7ff) { // 11 bit
205  *dest++ = static_cast<uint8>(((ch >> 6) & ((1 << 5) - 1)) | 0xc0);
206  *dest++ = static_cast<uint8>(((ch >> 0) & ((1 << 6) - 1)) | 0x80);
207  return 2;
208  }
209  if (ch <= 0xffff) { // 16 bit
210  *dest++ = static_cast<uint8>(((ch >> 12) & ((1 << 4) - 1)) | 0xc0);
211  *dest++ = static_cast<uint8>(((ch >> 6) & ((1 << 6) - 1)) | 0x80);
212  *dest++ = static_cast<uint8>(((ch >> 0) & ((1 << 6) - 1)) | 0x80);
213  return 3;
214  }
215  if (ch <= 0x10ffff) { // 21 bit
216  *dest++ = static_cast<uint8>(((ch >> 18) & ((1 << 3) - 1)) | 0xc0);
217  *dest++ = static_cast<uint8>(((ch >> 12) & ((1 << 6) - 1)) | 0x80);
218  *dest++ = static_cast<uint8>(((ch >> 6) & ((1 << 6) - 1)) | 0x80);
219  *dest++ = static_cast<uint8>(((ch >> 0) & ((1 << 6) - 1)) | 0x80);
220  return 4;
221  }
222  // TAG: add iso codes also
223  return 0;
224  }

239  {

617  static inline MemoryDiff UCS4ToWChar(wchar* dest, const ucs4* src, MemorySize size) noexcept
618  {

620  return 0;
621  }

623  return UCS4ToUTF16(reinterpret_cast<utf16*>(dest), src, size);
624  } else if (sizeof(wchar) == sizeof(ucs4)) {
625  if (dest) {
626  for (const auto end = src + size; src != end; ++dest, ++src) { // no code validation
627  *dest = *src;

632  BASSERT(!"Unsupported wchar.");
