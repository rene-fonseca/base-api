867 template<class TYPE>
868 class BuildObjectModel {
869 private:
870 
871  ObjectModel o;
872  Reference<ObjectModel::String> VALUE;
873  Reference<ObjectModel::String> LEFT;
874  Reference<ObjectModel::String> RIGHT;
875  StringOutputStream sos;
876 public:
877 
878  BuildObjectModel()
879  {
880  VALUE = o.createString("value");
881  LEFT = o.createString("left");
882  RIGHT = o.createString("right");
883  }
884 
885  Reference<ObjectModel::Object> build(const BinaryNode<TYPE>* node)
886  {
887  if (node) {
888  auto object = o.createObject();
889  object->setValue(VALUE, sos << node->getValue());
890  if (auto left = node->getLeft()) {
891  object->setValue(LEFT, build(left));
892  }
893  if (auto right = node->getRight()) {
894  object->setValue(RIGHT, build(right));
895  }
896  return object;
897  }
898  return nullptr;
899  }
900 };
