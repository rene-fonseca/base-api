380  void shuffle()
381  {
382  MemorySize n = size;
383  if (n <= 1) {
384  return; // nothing to do
385  }
386 
387  PrimitiveStackArray<Node*> nodes(n);
388  {
389  auto src = first;
390  for (MemorySize i = 0; i < n; ++i) { // fill buffer
391  nodes[i] = src;
392  src = src->getNext();
393  }
394  BASSERT(!src);
395  }
396  base::shuffle(&nodes[0], &nodes[0] + nodes.size());
397 
398  // rebuild node list
399  {
400  auto src = nodes.begin();
401  auto node = *src++;
402  node->setPrevious(nullptr);
403  node->setNext(nullptr);
404  first = node;
405  last = node;
406 
407  const auto end = nodes.end();
408  while (src != end) {
409  node = *src++;
410  node->setPrevious(last);
411  node->setNext(nullptr);
412  last->setNext(node);
413  last = node;
414  }
415  }
