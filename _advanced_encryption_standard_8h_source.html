129  inline unsigned int cipherPush(uint8* dest, const uint8* src, unsigned int size) noexcept
130  {
131  const unsigned int BLOCK_SIZE = CIPHER::BLOCK_SIZE;
132  const uint8* first = dest;
133  if ((size >= BLOCK_SIZE) ||
134  ((size + bytesInBuffer) >= BLOCK_SIZE)) {
135  CIPHER cipher;
136  if (bytesInBuffer > 0) { // do we need to empty internal buffer
137  copy(buffer + bytesInBuffer, src, BLOCK_SIZE - bytesInBuffer);
138  cipher(dest, buffer, schedule); // internal buffer
139  dest += BLOCK_SIZE;
140  src += BLOCK_SIZE - bytesInBuffer;
141  size -= BLOCK_SIZE - bytesInBuffer;
142  bytesInBuffer = 0;
143  }
144  while (size >= BLOCK_SIZE) {
145  cipher(dest, src, schedule); // external buffer
146  dest += BLOCK_SIZE;
147  src += BLOCK_SIZE;
148  size -= BLOCK_SIZE;
149  }
150  }
151  copy(buffer + bytesInBuffer, src, size);
152  bytesInBuffer += size;
153  return dest - first;
154  }

160  BASSERT(bytesInBuffer < BLOCK_SIZE);
