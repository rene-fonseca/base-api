60  void setValue(TYPE* _value, Counter* _references)
61  {
62  if (value) {
63  BASSERT(_value != value);
64  BASSERT(references);
65 #if !defined(__clang_analyzer__)
66  // assert(references);
67  if (!--*references) {
68  if (_value && !_references) {
69  ++*references; // reuse - 1 reference
70  value = _value;
71  return;
72  }
73  delete references; // less likely to throw
74  references = nullptr;
75  TYPE* oldValue = value;
76  value = nullptr;
77  delete oldValue; // could throw
78  } else {
79  value = nullptr;
80  references = nullptr;
81  }
82 #endif
83  }
84 
85  if (_value) {
86  if (_references) {
87  references = _references;
88  ++*references; // add reference
89  } else {
90  references = new Counter(1);
91  }
92  value = _value;
93  }
94  }
