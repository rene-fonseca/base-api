881 template<class TYPE>
882 FormatOutputStream& operator<<(
883  FormatOutputStream& stream, const Sequence<TYPE>& value)
884 {
885  FormatOutputStream::PushContext push(stream);
886  const TYPE* src = value.getValue();
887  const TYPE* end = src + value.getSize();
888  const Literal& separator = value.getSeparator();
889  if (separator.getLength() > 0) {
890  --end;
891  while (src < end) {
892  stream << *src << separator;
893  ++src;
894  }
895  if (src == end) { // if not empty
896  stream << *src;
897  }
898  } else {
899  while (src < end) {
900  stream << *src;
901  ++src;
902  }
903  }
904  return stream;
905 }
