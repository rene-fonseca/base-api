864 template<class TYPE>
865 FormatOutputStream& operator<<(
866  FormatOutputStream& stream, const Sequence<TYPE>& value)
867 {
868  FormatOutputStream::PushContext push(stream);
869  const TYPE* src = value.getValue();
870  const TYPE* end = src + value.getSize();
871  const Literal& separator = value.getSeparator();
872  if (separator.getLength() > 0) {
873  --end;
874  while (src < end) {
875  stream << *src << separator;
876  ++src;
877  }
878  if (src == end) { // if not empty
879  stream << *src;
880  }
881  } else {
882  while (src < end) {
883  stream << *src;
884  ++src;
885  }
886  }
887  return stream;
888 }
