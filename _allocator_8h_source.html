109  inline TYPE* resize(TYPE* buffer, const MemorySize newSize, const MemorySize originalSize)
110  {
111  // buffer can be nullptr
112  BASSERT((!buffer && !originalSize) || (buffer && originalSize));
113  if (!INLINE_ASSERT(newSize != originalSize)) {
114  return buffer;
115  }
116  if (originalSize > newSize) {
117  fill(buffer + newSize, originalSize - newSize, RELEASE_MEMORY);
118  }
119  const auto adjustedNewSize = getAdjustedSize(newSize);
120  auto result = Heap::resize(buffer, adjustedNewSize); // reallocates - so do NOT have initialized objects in memory
121  if (originalSize < adjustedNewSize) {
122  fill(result + originalSize, adjustedNewSize - originalSize, INIT_MEMORY);
123  }
124  return result;
125  }

223  static void destroy(TYPE* begin, const TYPE* _end, MemorySize& destroyedCount)
224  {
225  BASSERT(begin <= _end);
226  if (IsUninitializeable<TYPE>() || std::is_trivially_destructible<TYPE>()) {
227  destroyedCount += (_end - begin);
228  fill(begin, _end - begin, DESTROY_OBJECT);
229  return;
230  }
231 
232  TYPE* end = (_end - begin) + begin;
233  if (useSafeDestroy && !std::is_nothrow_destructible<TYPE>()) {
234  std::exception_ptr e;
235  while (begin != end) { // we destroy from the end
236  --end;
237  try {
238  ++destroyedCount; // before destroy!
239  // TAG: needs testing - MSC behavior unknown
240  end->~TYPE(); // this can throw - if only 1 exception object is unwound - std::terminate() could still be called
241  } catch (...) {
242  // we dont know which exception to rethrow so we just use first
243  if (!e) {
244  e = std::current_exception();
245  }
246  }
247  fill(end, 1, DESTROY_OBJECT);
248  }
249  if (e) {
250  destroyedCount = _end - begin;
251  std::rethrow_exception(e);
252  }
253  } else {
254  while (begin != end) { // we destroy from the end
255  --end;
256  ++destroyedCount; // before destroy!
257  end->~TYPE(); // this can throw // what if noexcept still throws
258  fill(end, 1, DESTROY_OBJECT);
259  }
260  }
261  }

306  static void initializeByMove(TYPE* dest, TYPE* src, const TYPE* end, bool& moved)
307  {
308  BASSERT(src <= end);
309  if (!IsUninitializeable<TYPE>() && !IsRelocateable<TYPE>()) {
310  initializeByMove(dest, src, end); // we move all first
311  moved = true; // from here we can recover to some degree by leaking src
312  // we cannot release heap since object can have indirect self references
313  destroy2(src, end); // now we destroy all
314  } else { // just copy memory
315  while (src != end) {
316  *dest++ = *src++; // must not throw
317  }
318  moved = true;

320  }

571  void setSizeImpl(MemorySize size, const TYPE* value)
572  {
573  if (size != this->size) {
574  if (IsUninitializeable<TYPE>() && IsRelocateable<TYPE>()) {
575  // no need to destroy or initialize elements
576  elements = resize(elements, size, this->size); // ok if this throws
577  if (value) { // fill new elements
578  if (size > this->size) { // we are increased array size
579  base::fill(elements + this->size, size - this->size, *value);
580  }
581  }
582  this->size = size;
583  } else { // non-primitive case
584  TYPE* temp = nullptr; // new buffer
585  if (size > this->size) { // extend array
586  // TAG: check if reallocatable TYPE
587  temp = tryResize(elements, size, this->size); // wont throw - elements pointer still good!
588  BASSERT(!temp || (temp == elements));
589  if (!temp) {
590  temp = allocate(size); // new array - ok if this throws here
591  }
592  } else if (!canResizeInplace() && (size > 0)) {
593  temp = allocate(size); // new array - ok if this throws here
594  }
595 
596  auto original = detach();
597  Leaky<TYPE> leaky(original); // TAG: can we add callback to notify parent class to e.g. update cached size
598 
599  if (size < original.size) { // are we about to reduce the array
600  bool reduced = false;
601  if (!temp) { // try inplace resize
602  reduced = true;
603  destroy2(original.buffer + size, original.buffer + original.size); // we cannot recover if this throws
604  temp = tryResize(original.buffer, size, original.size); // wont throw
605  BASSERT(!temp || (temp == original.buffer)); // reallocation NOT allowed - we still have objects initialized
606  if (temp || (size == 0)) {
607  attach(temp, size);
608  return;
609  }
610  }
611  if (!temp && (size > 0)) {
612  try {
613  temp = allocate(size); // new array
614  } catch (...) {
615  attach(original); // not modified
616  leaky.clear();
617  _rethrow;
618  }
619  }
620  initializeByMove(temp, original.buffer, original.buffer + size); // we still need to destroy
621  attach(temp, size);
622  destroy2(original.buffer, original.buffer + (reduced ? size : original.size)); // we cannot recover if this throws - we leak original buffer
623  release(original.buffer, original.size); // free previous array
624  } else { // array is to be expanded
625  if (temp != original.buffer) { // not if inplace resized
626  initializeByMove(temp, original.buffer, original.buffer + original.size); // we still need to destroy
627  }
628 
629  // if construction fails we cannot recover - we can set size to the last valid element and continue
630  if (value) {
631  initialize(temp + original.size, temp + size, *value); // copy construction
632  } else {
633  initialize(temp + original.size, temp + size); // default initialization of new objects
634  }
635  // see "fail construct" example - can we just set size of the successful initialized?
636  attach(temp, size); // now we are free to use elements again
637 
638  if (temp != original.buffer) { // not if inplace resized
639  destroy2(original.buffer, original.buffer + original.size); // we cannot recover if this throws - we leak original buffer
640  release(original.buffer, original.size); // free previous array
641  }
642  }
643  }
644  }

713  MemoryDiff garbageCollect()
714  {
715  if (!elements) {
716  return 0;
717  }
718  const auto capacity = Heap::getSize(elements);
719  if (size == 0) {
720  auto original = detach();
721  release(original.buffer, original.size); // ok if this throws
722  return capacity * sizeof(TYPE);
723  }
724  if (align(capacity * sizeof(TYPE)) > align(size * sizeof(TYPE))) {
725  if (canResizeInplace()) {
726  auto _elements = tryResize(elements, size, capacity);
727  if (_elements) {
728  BASSERT(_elements == elements);
729  return (capacity - Heap::getSize(elements)) * sizeof(TYPE);
730  }
731  }
732 
733  if (IsRelocateable<TYPE>()) {
734  elements = resize(elements, size, capacity); // ok if this throws
735  return (capacity - Heap::getSize(elements)) * sizeof(TYPE); // could be negative
736  }
737 
738  // std::is_move_constructible<TYPE>() && std::is_nothrow_move_constructible<TYPE>();
739 
740  Span<TYPE> temp(allocate(size), size); // ok if throws
741  auto original = detach();
742  Leaky<TYPE> leaky(original);
743  initializeByMove(temp.buffer, original.buffer, original.buffer + original.size);
744  attach(temp);
745  destroy2(original.buffer, original.buffer + original.size);
746  release(original.buffer, original.size);
747  return (capacity - Heap::getSize(temp.buffer)) * sizeof(TYPE);
748  }
749  return 0;
750  }
