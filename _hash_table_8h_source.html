234  HashTableImpl(MemorySize capacity)
235  {
236  capacity = maximum(capacity, MINIMUM_CAPACITY);
237  capacity = minimum(capacity, MAXIMUM_CAPACITY);
238  log2OfCapacity = Math::iLog2(capacity);
239  if ((static_cast<MemorySize>(1) << log2OfCapacity) < capacity) {
240  ++log2OfCapacity;
241  capacity *= 2;
242  }
243  this->capacity = capacity;
244  mask = capacity - 1;
245  size = 0;
246  table.setSize(capacity);
247  fill<Node*>(table.getElements(), table.getSize(), 0);
248  }

253  HashTableImpl(const HashTableImpl& copy)
254  : table(copy.capacity),
255  capacity(copy.capacity),
256  mask(copy.mask),
257  log2OfCapacity(copy.log2OfCapacity),
258  size(copy.size)
259  {
260  table.setSize(capacity); // entries set below
261 
262  // copy all buckets
263  const Node* const* src = copy.getBuckets();
264  const Node* const* end = src + copy.capacity;
265  Node** dest = getBuckets();
266  while (src != end) {
267  const Node* srcNode = *src++;
268  if (srcNode) {
269  Node* firstNode = new Node(
270  srcNode->getHash(),
271  srcNode->getKey(),
272  srcNode->getValue()
273  );
274  *dest++ = firstNode;
275  Node* destNode = firstNode;
276  srcNode = srcNode->getNext();
277  while (srcNode) {
278  destNode->setNext(
279  new Node(srcNode->getHash(), srcNode->getKey(), srcNode->getValue())
280  );
281  destNode = destNode->getNext();
282  srcNode = srcNode->getNext();
283  }
284  } else {
285  *dest++ = 0;
286  }
287  }

293  inline void grow()
294  {
295  if (capacity < MAXIMUM_CAPACITY) {
296  capacity *= 2;
297  mask = capacity - 1;
298  table.setSize(capacity); // upper entries set below
299  ++log2OfCapacity;
300 
301  // unfold buckets
302  Node** lowerBucket = getBuckets();
303  Node** upperBucket = lowerBucket + capacity/2;
304  const Node* const* endBucket = upperBucket;
305  const MemorySize bitMask = capacity/2; // the bit that desides the half
306  while (lowerBucket != endBucket) {
307  Node* srcNode = *lowerBucket;
308  if (srcNode) {
309  // keep order and do NOT construct or destroy nodes
310  if ((srcNode->getHash() & bitMask) == 0) {
311  // move from first to second bucket
312  Node* parentSrcNode = srcNode;
313  srcNode = srcNode->getNext();
314  while (srcNode && ((srcNode->getHash() & bitMask) == 0)) {
315  parentSrcNode = srcNode;
316  srcNode = srcNode->getNext();
317  }
318  if (srcNode) {
319  parentSrcNode->setNext(srcNode->getNext()); // unlink
320  *upperBucket = srcNode;
321  srcNode = srcNode->getNext();
322 
323  Node* destNode = *upperBucket;
324  // destNode->setNext(nullptr); // only set for last node
325 
326  while (srcNode) {
327  Node* nextNode = srcNode->getNext();
328  if (srcNode->getHash() & bitMask) {
329  parentSrcNode->setNext(nextNode); // unlink
330  destNode->setNext(srcNode);
331  destNode = destNode->getNext();
332  // destNode->setNext(nullptr); // only set for last node
333  }
334  parentSrcNode = srcNode;
335  srcNode = nextNode;
336  }
337  destNode->setNext(nullptr);
338  }
339  } else {
340  // move from second to first bucket
341  *upperBucket = *lowerBucket;
342  *lowerBucket = 0;
343 
344  Node* parentSrcNode = srcNode;
345  srcNode = srcNode->getNext();
346  while (srcNode && (srcNode->getHash() & bitMask)) {
347  parentSrcNode = srcNode;
348  srcNode = srcNode->getNext();
349  }
350  if (srcNode) {
351  parentSrcNode->setNext(srcNode->getNext()); // unlink
352  *lowerBucket = srcNode;
353  srcNode = srcNode->getNext();
354 
355  Node* destNode = *lowerBucket;
356  // destNode->setNext(nullptr); // only set for last node
357 
358  while (srcNode) {
359  Node* nextNode = srcNode->getNext();
360  if ((srcNode->getHash() & bitMask) == 0) {
361  parentSrcNode->setNext(nextNode); // unlink
362  destNode->setNext(srcNode);
363  destNode = destNode->getNext();
364  // destNode->setNext(nullptr); // only set for last node
365  }
366  parentSrcNode = srcNode;
367  srcNode = nextNode;
368  }
369  destNode->setNext(nullptr); // terminate linked list
370  }
371  }
372  } else {
373  *upperBucket = 0; // lower bucket already empty
374  }
375  ++lowerBucket;
376  ++upperBucket;
377  }
378  }
379  }

384  inline void shrink() noexcept
385  {
386  if (capacity > MINIMUM_CAPACITY) {
387  // fold upper half buckets into lower buckets
388  Node** lowerBucket = getBuckets();
389  Node** upperBucket = lowerBucket + capacity/2;
390  const Node* const* endBucket = upperBucket;
391  while (lowerBucket != endBucket) {
392  // merge buckets
393  if (!*upperBucket) {
394  // nothing to do
395  } else if (!*lowerBucket) {
396  *lowerBucket = *upperBucket;
397  } else {
398  // TAG: bucket could hold link to last node
399  Node* node = *lowerBucket;
400  while (node->getNext()) { // go to last node
401  node = node->getNext();
402  }
403  node->setNext(*upperBucket);
404  }
405  ++lowerBucket;
406  ++upperBucket;
407  }
408  capacity /= 2;
409  mask = capacity - 1;
410  --log2OfCapacity;
411  table.setSize(capacity);
412  }
413  }

434  bool hasKey(const Key& key) noexcept
435  {
436  const unsigned long hash = getHash(key);
437  Node** bucket = getBuckets() + (hash & mask);
438 
439  Node* parent = nullptr;
440  Node* child = *bucket;
441  if (child) {
442  if ((child->getHash() == hash) && (child->getKey() == key)) {
443  return true; // this case should be fairly likely
444  }
445  parent = child;
446  child = child->getNext();
447  while (child &&
448  ((child->getHash() != hash) || (child->getKey() != key))) {
449  parent = child;
450  child = child->getNext();
451  }
452  if (child) {
453  Node* temp = child->getNext();
454  Node* first = *bucket;
455  child->setNext(first); // cyclic loop to first
456  *bucket = child; // move to first - still cyclic loop
457  parent->setNext(temp); // detach child
458  }
459  }
460  return child;
461  }

523  Node* grandparent = nullptr;
524  Node* parent = nullptr;
525  Node* child = *bucket;

527  grandparent = parent;

531  if (child) {
532  if (grandparent) {
533  grandparent->setNext(child);

535  child->setNext(parent);

572  void add(const Key& key, const Value& value)
573  {
574  const unsigned long hash = getHash(key);
575  Node** buckets = getBuckets() + (hash & mask);
576  if (*buckets) {
577  Node* parent = nullptr;
578  Node* child = *buckets;
579  while (child && ((child->getHash() != hash) || (child->getKey() != key))) {
580  parent = child;
581  child = child->getNext();
582  }
583  if (child) { // found key
584  child->getValue() = value;
585  } else { // if value not already in list
586  child = new Node(hash, key, value);
587  if (parent) {
588  parent->setNext(child);
589  } else {
590  *buckets = child;
591  }
592  ++size;
593  }
594  } else {
595  *buckets = new Node(hash, key, value);
596  ++size;
597  }
598  if (size > capacity) { // what is the best criteria
599  grow();
600  }
601  }

630  {

632  Node** bucket = getBuckets() + (hash & mask);
633  Node* child = *bucket;
634  if (!child) {
635  _throw InvalidKey(this);
636  }

638  *bucket = child->getNext(); // unlink first node (next could be 0)
639  } else {
640  Node* parent = child;
641  child = child->getNext();
642  while (child && ((child->getHash() != hash) || (child->getKey() != key))) {
643  parent = child;

645  }
646  if (!child) {

648  }
649  parent->setNext(child->getNext()); // unlink node from linked list
650  }
