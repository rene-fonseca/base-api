475 void move(TYPE* dest, const TYPE* src, MemorySize count)
476 {
477  if (((dest < src) && ((dest + count) < src)) || // technically <= is allowed
478  ((src < dest) && ((src + count) < dest))) {
479  copy<TYPE>(dest, src, count); // use normal copy when no overlap
480  return;
481  }
482 
483  if (IsRelocateable<TYPE>()) {
484  move<uint8>(
485  reinterpret_cast<uint8*>(dest),
486  reinterpret_cast<const uint8*>(src),
487  sizeof(TYPE) * count
488  );
489  } else {
490  if (dest < src) {
491  const TYPE* const end = dest + count;
492  while (dest != end) {
493  *dest = *src;
494  ++dest;
495  ++src;
496  }
497  } else {
498  const TYPE* first = dest;
499  dest += count;
500  src += count;
501  while (dest > first) {
502  *--dest = *--src;
503  }
504  }

579  return;

584  case 6:

592  case 2:

594  case 1:

596  case 0:
597  break;
