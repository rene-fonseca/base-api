261  {
262  if (_value) { // skip if pointer is invalid
263  ReferenceCountedObject::ReferenceImpl(*_value).addReference();
264  }
265  TYPE* oldValue = value;
266  value = _value;
267  if (oldValue) { // skip if pointer is invalid
268  if (ReferenceCountedObject::ReferenceImpl(*oldValue).removeReference()) {
269  if (std::has_virtual_destructor<TYPE>()) {
270  destroyReference(oldValue); // could throw

272  if (!garbageCollectOptional(oldValue)) {
