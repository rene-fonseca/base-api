Detailed Description

template<class TYPE, class PREFIX, class SUFFIX>
class InvokeOutfix< TYPE, PREFIX, SUFFIX >

Outfix invocation.

Wrapper class used to invoke prefix and suffix pairs.

class MyClass {
public:
MyClass() noexcept {
}
int myMethod(int value) noexcept {
}
};
class MyLock {
public:
MyLock();
void lock();
void unlock();
};
class MyPrefix {
private:
MyLock& lock;
public:
MyPrefix(MyLock& _lock) noexcept : lock(_lock) {
}
inline operator()() {
lock.lock();
}
};
class MySuffix {
private:
MyLock& lock;
public:
MySuffix(MyLock& _lock) noexcept : lock(_lock) {
}
inline void operator()() {
lock.unlock();
}
};
void myFunction()
{
MyLock myLock;
InvokeOutfix outfix(object, MyPrefix(myLock), MySuffix(myLock));
int result = outfix->myMethod(1234);
}
