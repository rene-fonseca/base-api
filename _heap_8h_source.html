485 class PrimitiveStackArray {
486 private:
487 
488  TYPE stack[STACK_SIZE];
489  HeapBlockOwned<TYPE> heap;
490  MemorySize count = 0;
491  TYPE* buffer = nullptr;
492 
493  inline void update() noexcept
494  {
495  if (count == 0) {
496  buffer = nullptr; // not required in many cases - sometimes buffer is checked against nullptr
497  return;
498  }
499  buffer = (count <= STACK_SIZE) ? stack : heap;
500  }
501 public:
502 
504  PrimitiveStackArray(MemorySize size = 0) // we could use STACK_SIZE - but makes it harder to refactor later
505  {
506 #if defined(_COM_AZURE_DEV__BASE__ANY_DEBUG)
507  // do not initialize buffer for release build
508  clear(stack);
509 #endif
510  BASSERT(size <= STACK_SIZE);
511  if (size > 0) {
512  if (size > STACK_SIZE) {
513  heap.resize(size);
514  }
515  this->count = size;
516  update();
517  }
518  }
519 
521  inline bool isUsingStack() const noexcept
522  {
523  return count <= STACK_SIZE;
524  }
525 
527  inline bool isUsingHeap() const noexcept
528  {
529  return count > STACK_SIZE;
530  }
531 
533  void resize(MemorySize count)
534  {
535  if (count != this->count) {
536  // we never go back from heap to stack!
537  if (!((this->count <= STACK_SIZE) && (count <= STACK_SIZE))) { // use heap
538  heap.resize(count); // copies if already in use
539  if (this->count <= STACK_SIZE) {
540  copy<TYPE>(heap, stack, this->count);
541  }
542  }
543  this->count = count;
544  update();
545  }
546  }
