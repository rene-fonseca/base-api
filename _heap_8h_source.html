616 template<class TYPE, MemorySize STACK_SIZE = (sizeof(TYPE) <= 4096) ? 4096/sizeof(TYPE) : 1>
617 class PrimitiveStackArray {
618 private:
619 
620  TYPE stack[STACK_SIZE];
621  HeapBlockOwned<TYPE> heap;
622  MemorySize count = 0;
623  TYPE* buffer = nullptr;
624 
625  inline void update() noexcept
626  {
627  if (count == 0) {
628  buffer = nullptr; // not required in many cases - sometimes buffer is checked against nullptr
629  return;
630  }
631  buffer = (count <= STACK_SIZE) ? stack : heap;
632  }
633 public:
634 
636  PrimitiveStackArray(MemorySize size = 0) // we could use STACK_SIZE - but makes it harder to refactor later
637  {
638  BASSERT(size <= STACK_SIZE);
639  if (size > 0) {
640  if (size > STACK_SIZE) {
641  heap.resize(size);
642  }
643  this->count = size;
644  update();
645  }
646  }
647 
649  inline bool isUsingStack() const noexcept
650  {
651  return count <= STACK_SIZE;
652  }
653 
655  inline bool isUsingHeap() const noexcept
656  {
657  return count > STACK_SIZE;
658  }
659 
661  void resize(MemorySize count)
662  {
663  if (count != this->count) {
664  // we never go back from heap to stack!
665  if (!((this->count <= STACK_SIZE) && (count <= STACK_SIZE))) { // use heap
666  heap.resize(count); // copies if already in use
667  if (this->count <= STACK_SIZE) {
668  copy<TYPE>(heap, stack, this->count);
669  }
670  }
671  this->count = count;
672  update();
673  }
674  }
