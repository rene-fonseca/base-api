179 public:
180

182  {
183  owner = _owner;
184  modificationId = _modificationId;
185  }
186 
187  void ensureUnmodified() const
188  {
189  if (owner && (modificationId != owner->iteratorModificationId)) {
190  throw IteratorException("Iterator used after owner modification.");
191  }
192  }

194  static void ensureCompatible(const IteratorOwnerContext& a, const IteratorOwnerContext& b)
195  {
196  a.ensureUnmodified();
197  b.ensureUnmodified();
198  // nullptr allowed
199  if (a.owner != b.owner) {
200  throw IteratorException("Comparison of iterators belonging to different owners.");
201  }
202  }
