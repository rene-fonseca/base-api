37  static Pointer next(Pointer src, ucs4& ch)
38  {
39  const int status = Unicode::readUCS4(src, ch);
40  if (status > 0) {
41  src += status;
42  } else {
43  switch (status) {
44  case Unicode::ERROR_INCOMPLETE:
45  throw MultibyteException(MultibyteException::INCOMPLETE_CHARACTER_CODE);
46  case Unicode::ERROR_BAD_ENCODING:
47  throw MultibyteException(MultibyteException::INVALID_CHARACTER_CODE);
48  default:
49  throw MultibyteException();
50  }
51  }
52  return src;
53  }
