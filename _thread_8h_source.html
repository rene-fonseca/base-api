531 template<unsigned int SIZE>
532 class FixedResourceManager {
533 public:
534 
535  bool slots[SIZE]; // debug only
536  unsigned int availableSlots[SIZE] = {0}; // available slots
537  unsigned int count = SIZE; // number of available slots
538 
540  FixedResourceManager() noexcept
541  {
542  for (unsigned int i = 0; i < SIZE; ++i) {
543  slots[i] = true;
544  availableSlots[i] = i;
545  }
546  }
547 
549  int acquire() noexcept
550  {
551  if (count > 0) {
552  --count;
553  auto id = availableSlots[count];
554  BASSERT(slots[id]);
555  slots[id] = false; // mark in use
556  return id;
557  }
558  return -1;
559  }
560 
562  inline void release(unsigned int id) noexcept
563  {
564  BASSERT(id < SIZE);
565  BASSERT(!slots[id]); // must be allocated
566  slots[id] = true; // mark available
567  availableSlots[count++] = id; // return to available
568  }
569 };
