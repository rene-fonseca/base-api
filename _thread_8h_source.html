572 template<unsigned int SIZE>
573 class FixedResourceManager {
574 public:
575 
576  bool slots[SIZE]; // debug only
577  unsigned int availableSlots[SIZE] = {0}; // available slots
578  unsigned int count = SIZE; // number of available slots
579 
581  FixedResourceManager() noexcept
582  {
583  for (unsigned int i = 0; i < SIZE; ++i) {
584  slots[i] = true;
585  availableSlots[i] = i;
586  }
587  }
588 
590  int acquire() noexcept
591  {
592  if (count > 0) {
593  --count;
594  auto id = availableSlots[count];
595  BASSERT(slots[id]);
596  slots[id] = false; // mark in use
597  return id;
598  }
599  return -1;
600  }
601 
603  inline void release(unsigned int id) noexcept
604  {
605  BASSERT(id < SIZE);
606  BASSERT(!slots[id]); // must be allocated
607  slots[id] = true; // mark available
608  availableSlots[count++] = id; // return to available
609  }
610 };
